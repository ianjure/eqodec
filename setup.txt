## Conda Environment Setup (Manual):

conda create -n eqodec python=3.10 -y
conda activate eqodec
conda install ipykernel notebook -y
python -m ipykernel install --user --name=eqodec --display-name "eqodec"
pip install torch==2.4.1+cu118 torchvision==0.19.1+cu118 torchaudio==2.4.1+cu118 --extra-index-url https://download.pytorch.org/whl/cu118
conda install -c conda-forge ffmpeg av=10.0.0 opencv -y
pip install -r requirements.txt

## Conda Environment Setup (Yaml):

conda env create -f environment.yml
conda activate eqodec
python -m ipykernel install --user --name=eqodec --display-name "eqodec"
